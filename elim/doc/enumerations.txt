############################################################################

Fetching defined enumerations:

  (function-call nil (enumerations ((id . "666"))) (alist nil))

Or to fetch a single enum type:

  (function-call nil 
    (enumerations ((id . "667"))) 
    (alist nil 
      (string ((name . "enum-type")) ":blist-node-type")))
